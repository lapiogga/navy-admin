export interface BoardConfig {
  id: string
  boardName: string           // 게시판 이름 (예: '공지사항', '질의응답')

  description: string
  useCategory: boolean        // 카테고리 사용 여부

  useComment: boolean         // 댓글 허용 여부
  maxAttachSize: number       // MB 단위 최대 첨부 크기
  createdAt: string
  updatedAt: string
}

/** 게시판 카테고리 */
export interface BoardCategory {
  id: string
  boardId: string
  categoryName: string
  sortOrder: number
  useYn: string
}

/** 게시글 */
export interface BoardPost {
  id: string
  boardId: string
  categoryId?: string
  title: string
  content: string
  authorId: string
  authorName: string
  authorUnit: string
  viewCount: number
  isPinned: boolean           // 공지 고정 여부

  attachments: BoardAttachment[]
  createdAt: string
  updatedAt: string
}

export interface BoardAttachment {
  id: string
  postId: string
  fileName: string
  fileSize: number            // bytes

  uploadedAt: string
}

/** 댓글 */
export interface BoardComment {
  id: string
  postId: string
  content: string
  authorId: string
  authorName: string
  createdAt: string

export interface BoardAdmin {
  id: string
  boardId: string
  userId: string
  userName: string
  userRank: string
  assignedAt: string
}

/** 게시판 사용자 설정 */
export interface BoardUser {
  id: string
  boardId: string
  userId: string
  userName: string
  userRank: string

  assignedAt: string
}

}

**2) entities/board/api.ts 생성:**
7개 API 객체:

- `boardConfigApi`:
  - `list(params: PageRequest): Promise<PageResponse<BoardConfig>>`
  - `detail(id: string): Promise<ApiResult<BoardConfig>>`
  - `create(data: Omit<BoardConfig, 'id' | 'createdAt' | 'updatedAt'>)`
  - `update(id, data)`, `delete(id)`
  - 경로: `/common/board-configs`, `/common/board-configs/:id`

- `boardCategoryApi`:
  - `list(boardId: string): Promise<ApiResult<BoardCategory[]>>` (비페이지, 전체 반환)
  - `create(boardId, data)`, `update(boardId, id, data)`, `delete(boardId, id)`
  - 경로: `/common/boards/:boardId/categories`

- `boardPostApi`:
  - `list(boardId: string, params: PageRequest & { categoryId?: string; keyword?: string }): Promise<PageResponse<BoardPost>>`
  - `detail(boardId: string, id: string)`
  - `create(boardId, data)`, `update(boardId, id, data)`, `delete(boardId, id)`
  - 경로: `/common/boards/:boardId/posts`, `/common/boards/:boardId/posts/:id`

- `boardCommentApi`:
  - `list(postId: string): Promise<ApiResult<BoardComment[]>>` (비페이지, 전체 반환)
  - `create(postId, data: { content: string })`, `delete(postId, commentId)`
  - 경로: `/common/posts/:postId/comments`

- `boardAdminApi`:
  - `list(boardId: string): Promise<ApiResult<BoardAdmin[]>>` (비페이지)
  - `assign(boardId: string, userIds: string[]): Promise<ApiResult>`
  - `remove(boardId: string, adminId: string): Promise<ApiResult>`
  - 경로: `/common/boards/:boardId/admins`

- `boardUserApi`:
  - `list(boardId: string, params: PageRequest): Promise<PageResponse<BoardUser>>`
  - `assign(boardId: string, userIds: string[]): Promise<ApiResult>`
  - `remove(boardId: string, userId: string): Promise<ApiResult>`
  - 경로: `/common/boards/:boardId/users`

- `boardUnitApi`:
  - `list(boardId: string): Promise<ApiResult<BoardUnit[]>>` (비페이지)
  - `assign(boardId: string, units: {unitCode: string, unitName: string}[]): Promise<ApiResult>`
  - `remove(boardId: string, unitId: string): Promise<ApiResult>`
  - 경로: `/common/boards/:boardId/units`

- 첨부파일은 antd Upload의 action으로 직접 호출되므로 별도 API 객체 불필요. 삭제만:
  - `boardAttachmentApi.delete(id)`
  - 경로: DELETE `/common/board/attachments/:id`
