        CLIENT["apiClient (Axios)"]
        MSW["MSW 2 Service Worker<br/>네트워크 레벨 인터셉션"]

    end

    REACT --> ANTD
    REACT --> TAILWIND
    REACT --> ZUSTAND
    REACT --> TQUERY
    REACT --> ROUTER

    VITE --> REACT

    style FUTURE fill:#e2e8f0,stroke:#94a3b8,stroke-dasharray: 5 5
